CREATE TABLE TIPO_TRABAJADOR
(
	CODIGO_TIPO_TRABAJADOR SERIAL,
	NOMBRE VARCHAR(100),
	FLAG INTEGER,
	PRIMARY KEY(CODIGO_TIPO_TRABAJADOR)
);

CREATE TABLE EXPEDIENTE
(
	NUM_EXPEDIENTE INTEGER,
	CODIGO_PROCESO INTEGER,
	CODIGO_PERSONA INTEGER,
	DNI_TRABAJADOR INTEGER,
	TRABAJADOR_ABOGADO INTEGER,
	TRABAJADOR_RESPONSABLE INTEGER,
	TRABAJADOR_SECRETARIO VARCHAR(300),
	NUM_EXPEDIENTE_22DIGITOS VARCHAR(300),
	NUM_JUZGADO VARCHAR (300),
	CODIGO_CAUTELAR VARCHAR(300),
	FLAG INTEGER,
	FECHA_ASIGNACION DATE,
	PRIMARY KEY (NUM_EXPEDIENTE),
	FOREIGN KEY (CODIGO_PROCESO) REFERENCES PROCESO (CODIGO_PROCESO),
	FOREIGN KEY (CODIGO_PERSONA) REFERENCES PERSONA (CODIGO_PERSONA),
	FOREIGN KEY (DNI_TRABAJADOR) REFERENCES TRABAJADOR (DNI_TRABAJADOR)
);

CREATE TABLE PERSONA
(
	
	CODIGO_PERSONA SERIAL,
	CODIGO_DISTRITO INTEGER,
	DOI_CLIENTE INTEGER,
	NOMBRE_PERSONA VARCHAR (400),
	APELLIDO_PATERNO VARCHAR (300),
	APELLIDO_MATERNO VARCHAR (300),
	NOMBRE_CONYUGE VARCHAR (300),
	APELLIDO_PATERNO_CONYUGE VARCHAR (300),
	APELLIDO_MATERNO_CONYUGE VARCHAR (300),
	REPRESENTANTE VARCHAR (300),
	DOMICILIO_PERSONA VARCHAR (500),
	DIRECCION_ALTERNA VARCHAR (500),
	OPERACION VARCHAR (300),
	FLAG INTEGER,
	PRIMARY KEY (CODIGO_PERSONA),
	FOREIGN KEY (CODIGO_DISTRITO) REFERENCES DISTRITO(CODIGO_DISTRITO)		
);

CREATE TABLE TRABAJADOR
(
	DNI_TRABAJADOR INTEGER NOT NULL,
	CODIGO_TIPO_TRABAJADOR INTEGER,
	NOMBRE VARCHAR(50),
	APELLIDO_PATERNO VARCHAR (100),
	APELLIDO_MATERNO VARCHAR (100),
	CONTRA VARCHAR (300),
	FLAG INTEGER,
	PRIMARY KEY (DNI_TRABAJADOR),
	FOREIGN KEY (CODIGO_TIPO_TRABAJADOR) REFERENCES TIPO_TRABAJADOR (CODIGO_TIPO_TRABAJADOR)
);

CREATE TABLE CARTERA
(

	NOMBRE VARCHAR(200),
	FLAG INT,
	PRIMARY KEY (CODIGO_CARTERA)
);
CREATE TABLE CIUDAD
(
	CODIGO_CIUDAD SERIAL,
	NOMBRE VARCHAR(200),
	FLAG INT,
	PRIMARY KEY (CODIGO_CIUDAD)
);
CREATE TABLE ESTUDIO
(
	CODIGO_ESTUDIO SERIAL,
	NOMBRE VARCHAR(200),
	FLAG INT,
	PRIMARY KEY (CODIGO_ESTUDIO)
);
CREATE TABLE JUZGADO
(
	CODIGO_JUZGADO SERIAL,
	NOMBRE VARCHAR(200),
	FLAG INT,
	PRIMARY KEY (CODIGO_JUZGADO)
);
CREATE TABLE MONEDA
(
	CODIGO_MONEDA SERIAL,
	NOMBRE VARCHAR(200),
	PRIMARY KEY(CODIGO_MONEDA)
);

CREATE TABLE TRAMITE_DEVOLUCION
(
	CODIGO_TRAMITE_DEVOLUCION SERIAL,
	CODIO_CREDITO INTEGER,
	NOMBRE_TRAMITE VARCHAR(300),
	FECHA_TRAMITE DATE,
	PRIMARY KEY (CODIGO_TRAMITE),
	FOREIGN KEY (CODIGO_CREDITO) REFERENCES CREDITO (CODIGO_CREDITO)
);

CREATE TABLE CREDITO
(
	CODIGO_CREDITO SERIAL,
	NUM_EXPEDIENTE INTEGER,
	CODIGO_CARTERA INTEGER,
	CODIGO_MONEDA INTEGER,
	CODIGO_CIUDAD INTEGER,
	CODIGO_ESTUDIO INTEGER,
	CODIGO_JUZGADO INTEGER,
	CASTIGOS VARCHAR (300),
	MODELO VARCHAR (300),
	PRODUCTO_MODULO VARCHAR (300),
	CDR VARCHAR(300),
	CUENTA_BT VARCHAR (100),
	CAPITAL VARCHAR (100),
	FECHA_PROTESTO DATE,
	TRAMITE_DEVOLUCION VARCHAR(100),
	OBSERVACION_DEVOLUCION VARCHAR (300),
	FECHA_TRAMITE_DEVOLUCION VARCHAR(100),
	FLAG INTEGER,
	PRIMARY KEY (CODIGO_CREDITO),
	FOREIGN KEY (NUM_EXPEDIENTE) REFERENCES EXPEDIENTE (NUM_EXPEDIENTE),
	FOREIGN KEY (CODIGO_CARTERA) REFERENCES CARTERA (CODIGO_CARTERA),
	FOREIGN KEY (CODIGO_MONEDA) REFERENCES MONEDA (CODIGO_MONEDA),
	FOREIGN KEY (CODIGO_CIUDAD) REFERENCES CIUDAD (CODIGO_CIUDAD),
	FOREIGN KEY (CODIGO_ESTUDIO) REFERENCES ESTUDIO (CODIGO_ESTUDIO),
	FOREIGN KEY (CODIGO_JUZGADO) REFERENCES JUZGADO (CODIGO_JUZGADO)
);
